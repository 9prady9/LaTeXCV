\newcommand{\skill}[2]{
	\pgfmathsetmacro\pgfxa{#2+1}
	\tikzstyle{scorestars}=[circle, draw, inner sep=0.32em,anchor=west]

	#1 \hspace*{\fill} \begin{tikzpicture}
		\foreach \i in {1,...,5} { % replace end value with #3 if ranking order has to be custom
			\pgfmathparse{(\i<=#2?"aforange!70":"white")}
			\edef\starcolor{\pgfmathresult}
			\draw (\i*1.3em,0) node[name=star\i,scorestars,fill=\starcolor]  {};
		}
		\pgfmathparse{(#2>int(#2)?int(#2+1):0}
		\let\partstar=\pgfmathresult
		\ifnum\partstar>0
		\pgfmathsetmacro\starpart{#2-(int(#2))}
		\path [clip] (star\partstar.north west) rectangle 
		($(star\partstar.south west)!\starpart!(star\partstar.south east)$);
		\fill (\partstar*1.3em,0) node[scorestars,fill=aforange!70]  {};
		\fi,
	\end{tikzpicture}
\newline
\vspace{-10pt}
}
